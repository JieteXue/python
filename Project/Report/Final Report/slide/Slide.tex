\section{Project Overview}
\begin{frame}{Project Overview}
  \begin{block}{Objective}
    Develop an interactive platform for playing and analyzing Impartial Combinatorial Games (ICGs) with automated winning strategy computation.
  \end{block}
  
  \begin{block}{Key Features}
    \begin{itemize}
      \item Support multiple classic ICG games (Nim, Kayles, etc.)
      \item Real-time strategy analysis and optimal move suggestions
      \item Interactive graphical user interface
      \item Extensible framework for new games
    \end{itemize}
  \end{block}
\end{frame}

\section{Technical Challenges}
\begin{frame}{Technical Challenges}
  \begin{itemize}[<+- | alert@+>]
    \item \textbf{State Space Explosion}: Exponential growth of possible states
    \item \textbf{Real-Time Requirements}: Need for sub-second response time
    \item \textbf{General Framework Design}: Supporting diverse game rules
    \item \textbf{UI/UX Design}: Creating intuitive interface with no prior experience
  \end{itemize}
\end{frame}

\section{Our Solutions}
\begin{frame}{System Architecture}
  \begin{block}{Three-Tier Architecture}
    \begin{itemize}
      \item \textbf{Game Logic Layer} \\Implements rules and state transitions for each game
      \item \textbf{AI Engine Layer} \\Optimized recursion with memoization for strategy analysis
      \item \textbf{User Interface Layer} \\Pygame-based interactive graphical interface
    \end{itemize}
  \end{block}
  
  \begin{block}{Modular Design}
    \begin{itemize}
      \item Each game implements standardized interfaces
      \item Core engine independent of specific game logic
      \item Easy to add new games
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Core Algorithm}
  \begin{block}{Optimized Recursion with Memoization}
    \vspace{-0.5em}
    \begin{align*}
      \text{win}(S) & = 
      \begin{cases} 
        \text{False} & \text{if } S \text{ is terminal} \\
        \lnot \bigwedge_{S' \in \text{moves}(S)} \text{win}(S') & \text{otherwise}
      \end{cases}
    \end{align*}
    \vspace{-1.5em}
  \end{block}
  
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Key Optimizations}
        \begin{itemize}
          \item \textbf{State Hashing}: Efficient state representation
          \item \textbf{Memoization}: Avoid redundant calculations
          \item \textbf{Early Termination}: Prune search space
        \end{itemize}
      \end{block}
    \end{column}
    
    \begin{column}{0.48\textwidth}
      \begin{block}{Performance}
        \begin{itemize}
          \item Solves medium states in $<1$ second
          \item Scales to complex game configurations
          \item Memory efficient
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
